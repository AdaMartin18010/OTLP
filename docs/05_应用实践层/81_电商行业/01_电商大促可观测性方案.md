# 电商大促可观测性方案

> **适用场景**: 电商平台大促、秒杀、抢购等高并发场景
> **技术难度**: ⭐⭐⭐⭐ (中高级)
> **最后更新**: 2026-03-17

---

## 1. 电商大促特点与挑战

### 1.1 流量特征

```
电商大促流量特征:

时间维度:
├── 预热期: 流量2-3倍增长
├── 开场期: 流量10-100倍激增
├── 平稳期: 维持高位
└── 收尾期: 快速回落

空间维度:
├── 商品详情页: 读多写少，缓存密集
├── 购物车: 高频读写，状态复杂
├── 订单系统: 强一致性，事务密集
├── 支付系统: 关键路径，零容错
└── 库存系统: 高并发扣减，热点问题

技术挑战:
├── 流量不确定性
├── 服务依赖复杂
├── 故障定位困难
├── 数据量爆炸
└── 实时性要求高
```

### 1.2 可观测性需求

| 场景 | 关键指标 | 延迟要求 | 采样策略 |
|------|----------|----------|----------|
| 首页访问 | QPS/错误率/延迟 | <100ms | 1%采样 |
| 商品搜索 | 查询延迟/结果数 | <200ms | 5%采样 |
| 下单流程 | 成功率/库存一致 | <500ms | 100%错误 |
| 支付回调 | 成功率/幂等性 | <1s | 100%全采 |
| 库存扣减 | 并发数/超卖率 | <50ms | 100%全采 |

### 2.2 分层采样策略

```yaml
# 大促分层采样配置
# flash-sale-sampling.yaml

processors:
  # 第一层: 网关Collector - 快速减负
  probabilistic_sampler/gateway:
    sampling_percentage: 1.0  # 1%头部采样

  # 第二层: 聚合Collector - 业务采样
  tail_sampling/aggregation:
    decision_wait: 5s
    num_traces: 500000  # 大促期间增加缓冲
    policies:
      # 关键业务100%采样
      - name: critical_business
        type: string_attribute
        string_attribute:
          key: business.type
          values: [payment, order_create, stock_deduct]

      # 错误全采
      - name: errors
        type: status_code
        status_code: {status_codes: [ERROR]}

      # 慢请求采样
      - name: slow_requests
        type: latency
        latency: {threshold_ms: 500}

      # 概率兜底
      - name: fallback
        type: probabilistic
        probabilistic: {sampling_percentage: 0.1}

  # 第三层: 按服务定制
  filter/service_specific:
    traces:
      span:
        - 'attributes["service.name"] == "stock-service" and status.code == 0'
        # 库存服务只采错误
```

## 3. 关键业务链路监控

### 3.1 下单链路追踪

```python
# 电商下单链路完整追踪示例
from opentelemetry import trace
from opentelemetry.trace import Status, StatusCode

class OrderService:
    def __init__(self):
        self.tracer = trace.get_tracer(__name__)

    def create_order(self, user_id, items, promotion_id=None):
        """
        完整下单链路追踪
        """
        with self.tracer.start_as_current_span(
            "order.create",
            attributes={
                "user.id": user_id,
                "item.count": len(items),
                "promotion.id": promotion_id or "none",
                "business.type": "order_create",  # 用于采样标记
                "business.priority": "high"
            }
        ) as span:
            try:
                # 1. 库存预占
                with self.tracer.start_as_current_span("stock.pre_reserve") as s1:
                    stock_result = self.stock_service.pre_reserve(items)
                    s1.set_attribute("stock.reserved", stock_result.success)
                    if not stock_result.success:
                        raise InsufficientStockError()

                # 2. 价格计算
                with self.tracer.start_as_current_span("price.calculate") as s2:
                    price = self.price_service.calculate(items, promotion_id)
                    s2.set_attribute("order.amount", price.total)
                    s2.set_attribute("discount.amount", price.discount)

                # 3. 创建订单
                with self.tracer.start_as_current_span("order.persist") as s3:
                    order = self.order_repository.create({
                        "user_id": user_id,
                        "items": items,
                        "amount": price.total,
                        "status": "created"
                    })
                    s3.set_attribute("order.id", order.id)

                # 4. 支付初始化
                with self.tracer.start_as_current_span("payment.init") as s4:
                    payment = self.payment_service.init(order)
                    s4.set_attribute("payment.id", payment.id)
                    s4.set_attribute("payment.timeout", payment.expire_time)

                # 5. 发送通知
                with self.tracer.start_as_current_span("notification.send") as s5:
                    self.notification_service.send_order_created(user_id, order)

                span.set_attribute("order.id", order.id)
                span.set_status(Status(StatusCode.OK))
                return order

            except InsufficientStockError as e:
                span.record_exception(e)
                span.set_attribute("error.type", "INSUFFICIENT_STOCK")
                span.set_status(Status(StatusCode.ERROR, "库存不足"))
                raise
            except Exception as e:
                span.record_exception(e)
                span.set_status(Status(StatusCode.ERROR, str(e)))
                # 库存回滚
                self.stock_service.release(items)
                raise
```

### 3.2 库存扣减监控

```python
# 库存系统高并发监控
class StockService:
    def __init__(self):
        self.tracer = trace.get_tracer(__name__)
        self.meter = metrics.get_meter(__name__)

        # 库存相关指标
        self.deduct_counter = self.meter.create_counter(
            "stock.deduct.total",
            description="Total stock deduction attempts"
        )
        self.deduct_latency = self.meter.create_histogram(
            "stock.deduct.duration_ms",
            description="Stock deduction latency"
        )
        self.conflict_counter = self.meter.create_counter(
            "stock.conflict.total",
            description="Optimistic lock conflicts"
        )

    async def deduct_stock(self, sku_id, quantity, order_id):
        with self.tracer.start_as_current_span(
            "stock.deduct",
            attributes={
                "sku.id": sku_id,
                "stock.quantity": quantity,
                "order.id": order_id,
                "business.type": "stock_deduct",
                "business.criticality": "high"
            }
        ) as span:
            start = time.time()

            try:
                # 使用Redis原子操作
                result = await self.redis.decrby(
                    f"stock:{sku_id}", quantity
                )

                latency = (time.time() - start) * 1000

                if result < 0:
                    # 库存不足，回滚
                    await self.redis.incrby(f"stock:{sku_id}", quantity)
                    span.set_attribute("stock.result", "INSUFFICIENT")
                    span.set_status(Status(StatusCode.ERROR, "库存不足"))
                    self.deduct_counter.add(1, {"result": "failed"})
                    return False

                span.set_attribute("stock.result", "SUCCESS")
                span.set_attribute("stock.remaining", result)
                self.deduct_counter.add(1, {"result": "success"})
                self.deduct_latency.record(latency)

                return True

            except Exception as e:
                span.record_exception(e)
                span.set_attribute("stock.result", "ERROR")
                self.deduct_counter.add(1, {"result": "error"})
                raise
```

### 4.2 容量计算示例

```python
# 大促容量计算器
class FlashSaleCapacityPlanner:
    def __init__(self):
        self.base_qps = 10000  # 平时QPS
        self.peak_multiplier = 50  # 大促峰值倍数

    def calculate(self):
        peak_qps = self.base_qps * self.peak_multiplier

        # Trace生成估算
        spans_per_request = 30  # 平均每个请求30个span
        raw_sps = peak_qps * spans_per_request  # 15M spans/s

        # 采样后 (1%头部 + 错误全采，假设错误率0.1%)
        sampled_sps = raw_sps * 0.01 + raw_sps * 0.001

        # Collector需求
        # 每个Collector处理10万 spans/s
        collectors_needed = int(sampled_sps / 100000) + 1

        # 内存需求 (批处理缓冲)
        memory_per_collector = 4  # GB

        # 网络带宽
        avg_span_size = 500  # bytes
        bandwidth_gbps = (sampled_sps * avg_span_size * 8) / 1e9

        return {
            "peak_qps": peak_qps,
            "raw_sps": raw_sps,
            "sampled_sps": sampled_sps,
            "collectors_needed": collectors_needed,
            "total_memory_gb": collectors_needed * memory_per_collector,
            "bandwidth_gbps": bandwidth_gbps,
            "recommendations": [
                f"部署 {collectors_needed} 个Collector实例",
                f"每个Collector配置 4核8G",
                f"预留 {bandwidth_gbps:.2f} Gbps 网络带宽",
                "启用分层采样策略",
                "配置HPA自动扩缩容"
            ]
        }

# 运行计算
planner = FlashSaleCapacityPlanner()
result = planner.calculate()
for rec in result["recommendations"]:
    print(f"✓ {rec}")
```

## 6. 故障应急

### 6.1 快速降级策略

```yaml
# 可观测性降级策略
# observability-degradation.yaml

processors:
  # 正常模式
  tail_sampling/normal:
    decision_wait: 10s
    policies: [errors, slow, probabilistic_10]

  # 降级模式 - 仅采错误
  tail_sampling/degraded:
    decision_wait: 5s
    policies: [errors_only]

  # 紧急模式 - 全丢弃
  filter/emergency:
    traces:
      span:
        - 'true'  # 丢弃所有

# 动态切换脚本
# switch_mode.sh
#!/bin/bash
MODE=${1:-normal}

kubectl patch configmap collector-config \
  --patch '{"data":{"config.yaml":"'$(cat config-$MODE.yaml | base64)'"}}'

kubectl rollout restart deployment collector-gateway
```

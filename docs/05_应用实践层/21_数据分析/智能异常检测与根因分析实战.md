### 1.2 异常类型分类

```yaml
# 可观测性数据中的异常类型

Point Anomalies (点异常):
  定义: 单个数据点与整体模式不符
  示例:
    - 错误率突然从0.1%跳到50%
    - 延迟突然从100ms跳到10s
  检测方法: 统计阈值、孤立森林
  难度: ⭐⭐ (容易)

Contextual Anomalies (上下文异常):
  定义: 在特定上下文中异常，但全局正常
  示例:
    - 凌晨2点的流量与白天相同 (正常情况下应该很低)
    - 周末的交易量与工作日下午相同
  检测方法: 时间序列分解、周期性分析
  难度: ⭐⭐⭐ (中等)

Collective Anomalies (集体异常):
  定义: 一组相关数据点的异常模式
  示例:
    - 多个服务同时出现延迟增加
    - 调用链上多个Span的异常延迟分布
  检测方法: 相关性分析、图神经网络
  难度: ⭐⭐⭐⭐ (困难)

Concept Drift (概念漂移):
  定义: 数据的统计特性随时间缓慢变化
  示例:
    - 用户增长导致基线流量缓慢上升
    - 系统老化导致响应时间逐渐增加
  检测方法: 漂移检测算法、在线学习
  难度: ⭐⭐⭐⭐⭐ (非常困难)
```

## 2. 统计方法

### 2.1 基于阈值的方法

```python
# 基础统计异常检测
import numpy as np
from scipy import stats

class StatisticalAnomalyDetector:
    """基于统计的异常检测器"""

    def __init__(self, method='zscore', threshold=3.0):
        self.method = method
        self.threshold = threshold
        self.history = []

    def fit(self, data):
        """学习历史数据的统计特性"""
        self.history = np.array(data)
        self.mean = np.mean(self.history)
        self.std = np.std(self.history)

        if self.method == 'iqr':
            self.q1 = np.percentile(self.history, 25)
            self.q3 = np.percentile(self.history, 75)
            self.iqr = self.q3 - self.q1

    def predict(self, value):
        """预测是否为异常"""
        if self.method == 'zscore':
            # Z-Score方法
            zscore = abs(value - self.mean) / self.std
            is_anomaly = zscore > self.threshold
            score = zscore

        elif self.method == 'iqr':
            # IQR方法 (对异常值更鲁棒)
            lower_bound = self.q1 - 1.5 * self.iqr
            upper_bound = self.q3 + 1.5 * self.iqr
            is_anomaly = value < lower_bound or value > upper_bound
            score = max(
                abs(value - self.q3) / self.iqr,
                abs(value - self.q1) / self.iqr
            )

        elif self.method == 'mad':
            # MAD方法 (中位数绝对偏差)
            median = np.median(self.history)
            mad = np.median(np.abs(self.history - median))
            modified_zscore = 0.6745 * (value - median) / mad
            is_anomaly = abs(modified_zscore) > self.threshold
            score = abs(modified_zscore)

        return {
            'is_anomaly': is_anomaly,
            'score': score,
            'value': value,
            'threshold': self.threshold
        }

# 使用示例
detector = StatisticalAnomalyDetector(method='zscore', threshold=3.0)

# 历史数据 (过去7天的错误率)
historical_error_rates = [0.1, 0.2, 0.1, 0.3, 0.2, 0.1, 0.2]
detector.fit(historical_error_rates)

# 检测当前值
current_error_rate = 5.0  # 突然跳到5%
result = detector.predict(current_error_rate)

print(f"是否异常: {result['is_anomaly']}")  # True
print(f"异常分数: {result['score']:.2f}")   # 约15.0 (远大于阈值3)
```

### 2.2 时间序列分解

```python
# 时间序列分解 (STL分解)
import numpy as np
from statsmodels.tsa.seasonal import STL
import pandas as pd

class TimeSeriesAnomalyDetector:
    """基于时间序列分解的异常检测"""

    def __init__(self, period=24):  # 默认24小时周期
        self.period = period

    def decompose(self, data, timestamps):
        """分解时间序列"""
        series = pd.Series(data, index=pd.to_datetime(timestamps))

        # STL分解: Seasonal and Trend decomposition using Loess
        stl = STL(series, period=self.period, robust=True)
        result = stl.fit()

        return {
            'trend': result.trend.values,
            'seasonal': result.seasonal.values,
            'residual': result.resid.values,
            'observed': result.observed.values
        }

    def detect_anomalies(self, data, timestamps, threshold=3.0):
        """基于残差检测异常"""
        decomposition = self.decompose(data, timestamps)
        residual = decomposition['residual']

        # 计算残差的统计特性
        mean_resid = np.mean(residual)
        std_resid = np.std(residual)

        anomalies = []
        for i, (ts, val, resid) in enumerate(zip(timestamps, data, residual)):
            zscore = abs(resid - mean_resid) / std_resid
            if zscore > threshold:
                anomalies.append({
                    'timestamp': ts,
                    'value': val,
                    'residual': resid,
                    'zscore': zscore,
                    'type': self._classify_anomaly(
                        val,
                        decomposition['trend'][i],
                        decomposition['seasonal'][i]
                    )
                })

        return anomalies

    def _classify_anomaly(self, value, trend, seasonal):
        """分类异常类型"""
        expected = trend + seasonal
        deviation = value - expected

        if abs(deviation) < expected * 0.1:
            return 'minor'
        elif abs(deviation) < expected * 0.5:
            return 'moderate'
        else:
            return 'severe'

# 使用示例
detector = TimeSeriesAnomalyDetector(period=24)  # 24小时周期

# 模拟数据: 一周的错误率，每小时一个点
# 正常模式: 工作日白天高，夜晚和周末低
np.random.seed(42)
n_days = 7
hours = n_days * 24

# 生成基准数据
base_pattern = []
for day in range(n_days):
    for hour in range(24):
        # 工作日(0-4)且工作时间(9-18)错误率较高
        if day < 5 and 9 <= hour <= 18:
            base = 0.5 + np.random.normal(0, 0.1)
        else:
            base = 0.1 + np.random.normal(0, 0.05)
        base_pattern.append(max(0, base))

# 在第3天插入异常
base_pattern[3 * 24 + 14] = 5.0  # 错误率突增到5%

timestamps = pd.date_range('2024-01-01', periods=hours, freq='H')

# 检测异常
anomalies = detector.detect_anomalies(base_pattern, timestamps)

for anomaly in anomalies:
    print(f"异常时间: {anomaly['timestamp']}")
    print(f"错误率: {anomaly['value']:.2f}%")
    print(f"异常类型: {anomaly['type']}")
```

### 2.3 孤立森林

```python
# 孤立森林 (Isolation Forest)
from sklearn.ensemble import IsolationForest
import numpy as np

class IsolationForestDetector:
    """基于孤立森林的异常检测"""

    def __init__(self, contamination=0.1, random_state=42):
        self.model = IsolationForest(
            contamination=contamination,  # 预期的异常比例
            random_state=random_state,
            n_estimators=100
        )

    def fit(self, X):
        """训练模型"""
        self.model.fit(X)

    def predict(self, X):
        """预测异常"""
        # -1 表示异常, 1 表示正常
        predictions = self.model.predict(X)
        # 转换为异常分数 (越小越异常)
        scores = self.model.decision_function(X)

        results = []
        for pred, score in zip(predictions, scores):
            results.append({
                'is_anomaly': pred == -1,
                'anomaly_score': -score,  # 转换为越大越异常
                'confidence': abs(score)
            })
        return results

# 多维异常检测示例
# 同时考虑多个指标

# 生成多维数据
np.random.seed(42)
n_samples = 1000

# 正常数据
normal_data = np.random.multivariate_normal(
    mean=[100, 0.5, 50],  # [延迟, 错误率, QPS]
    cov=[[10, 0, 5], [0, 0.01, 0], [5, 0, 20]],
    size=n_samples
)

# 添加一些异常
anomalies = np.array([
    [500, 0.5, 50],    # 延迟异常高
    [100, 5.0, 50],    # 错误率异常高
    [100, 0.5, 200],   # QPS异常高
    [500, 5.0, 200],   # 多维度异常
])

data = np.vstack([normal_data, anomalies])

# 训练和应用模型
detector = IsolationForestDetector(contamination=0.05)
detector.fit(data)

results = detector.predict(data)

# 统计异常
anomaly_count = sum(1 for r in results if r['is_anomaly'])
print(f"检测到 {anomaly_count} 个异常点")

# 查看最后几个点 (我们添加的异常)
for i, result in enumerate(results[-5:]):
    print(f"样本 {n_samples + i}: 异常={result['is_anomaly']}, 分数={result['anomaly_score']:.3f}")
```

## 3. 机器学习方法

### 3.1 LSTM时序预测

```python
# LSTM异常检测
import torch
import torch.nn as nn
import numpy as np

class LSTMAnomalyDetector(nn.Module):
    """基于LSTM的时序异常检测"""

    def __init__(self, input_size=1, hidden_size=64, num_layers=2):
        super().__init__()
        self.hidden_size = hidden_size
        self.num_layers = num_layers

        self.lstm = nn.LSTM(
            input_size=input_size,
            hidden_size=hidden_size,
            num_layers=num_layers,
            batch_first=True,
            dropout=0.2
        )

        self.fc = nn.Linear(hidden_size, input_size)

    def forward(self, x):
        lstm_out, _ = self.lstm(x)
        predictions = self.fc(lstm_out)
        return predictions

class LSTMAnomalyDetectorTrainer:
    """LSTM异常检测训练器"""

    def __init__(self, sequence_length=60):
        self.sequence_length = sequence_length
        self.model = None
        self.threshold = None

    def prepare_data(self, data):
        """准备序列数据"""
        sequences = []
        targets = []

        for i in range(len(data) - self.sequence_length):
            seq = data[i:i + self.sequence_length]
            target = data[i + 1:i + self.sequence_length + 1]
            sequences.append(seq)
            targets.append(target)

        return torch.FloatTensor(sequences), torch.FloatTensor(targets)

    def train(self, data, epochs=50):
        """训练模型"""
        X, y = self.prepare_data(data)

        # 添加特征维度
        X = X.unsqueeze(-1)
        y = y.unsqueeze(-1)

        self.model = LSTMAnomalyDetector()
        criterion = nn.MSELoss()
        optimizer = torch.optim.Adam(self.model.parameters(), lr=0.001)

        self.model.train()
        for epoch in range(epochs):
            optimizer.zero_grad()
            outputs = self.model(X)
            loss = criterion(outputs, y)
            loss.backward()
            optimizer.step()

            if (epoch + 1) % 10 == 0:
                print(f'Epoch [{epoch+1}/{epochs}], Loss: {loss.item():.6f}')

        # 计算异常阈值
        self.model.eval()
        with torch.no_grad():
            predictions = self.model(X)
            errors = torch.abs(predictions - y).numpy()
            self.threshold = np.percentile(errors, 95)  # 95百分位作为阈值

    def detect(self, sequence):
        """检测异常"""
        if self.model is None:
            raise ValueError("Model not trained")

        self.model.eval()
        with torch.no_grad():
            seq_tensor = torch.FloatTensor(sequence).unsqueeze(0).unsqueeze(-1)
            prediction = self.model(seq_tensor)

            error = torch.abs(prediction - seq_tensor).numpy().mean()
            is_anomaly = error > self.threshold

            return {
                'is_anomaly': is_anomaly,
                'error': error,
                'threshold': self.threshold,
                'confidence': min(error / self.threshold, 2.0)
            }

# 使用示例
trainer = LSTMAnomalyDetectorTrainer(sequence_length=60)

# 训练数据 (正常模式)
np.random.seed(42)
train_data = np.sin(np.linspace(0, 20*np.pi, 1000)) + np.random.normal(0, 0.1, 1000)

trainer.train(train_data, epochs=30)

# 测试数据 (包含异常)
test_data = np.sin(np.linspace(0, 2*np.pi, 100)) + np.random.normal(0, 0.1, 100)
test_data[50:55] = 5.0  # 插入异常

# 检测
for i in range(len(test_data) - 60):
    seq = test_data[i:i+60]
    result = trainer.detect(seq)

    if result['is_anomaly']:
        print(f"时间 {i+60}: 检测到异常, 误差={result['error']:.3f}")
```

### 3.2 Prophet时序预测

```python
# Prophet异常检测
from prophet import Prophet
import pandas as pd

class ProphetAnomalyDetector:
    """基于Prophet的异常检测"""

    def __init__(self, interval_width=0.95):
        self.model = Prophet(
            interval_width=interval_width,  # 置信区间
            yearly_seasonality=True,
            weekly_seasonality=True,
            daily_seasonality=True
        )
        self.interval_width = interval_width

    def fit(self, df):
        """
        训练模型
        df: DataFrame with 'ds' (datetime) and 'y' (value) columns
        """
        self.model.fit(df)

        # 训练集预测，计算残差分布
        forecast = self.model.predict(df)
        self.residual_mean = (df['y'] - forecast['yhat']).mean()
        self.residual_std = (df['y'] - forecast['yhat']).std()

    def detect(self, df):
        """检测异常"""
        forecast = self.model.predict(df)

        results = []
        for i, row in df.iterrows():
            actual = row['y']
            predicted = forecast.loc[i, 'yhat']
            lower_bound = forecast.loc[i, 'yhat_lower']
            upper_bound = forecast.loc[i, 'yhat_upper']

            # 判断是否在置信区间内
            is_anomaly = actual < lower_bound or actual > upper_bound

            # 计算异常分数
            residual = actual - predicted
            zscore = abs(residual - self.residual_mean) / self.residual_std

            results.append({
                'timestamp': row['ds'],
                'actual': actual,
                'predicted': predicted,
                'lower_bound': lower_bound,
                'upper_bound': upper_bound,
                'is_anomaly': is_anomaly,
                'residual': residual,
                'zscore': zscore
            })

        return pd.DataFrame(results)

# 使用示例: 检测API延迟异常

# 准备数据
np.random.seed(42)
dates = pd.date_range('2024-01-01', periods=30*24, freq='H')  # 30天，每小时

# 生成具有日周期和趋势的数据
base = 100
trend = np.linspace(0, 20, len(dates))
daily_pattern = 20 * np.sin(2 * np.pi * np.arange(len(dates)) / 24)
noise = np.random.normal(0, 5, len(dates))

latency = base + trend + daily_pattern + noise

# 插入异常 (第15天的异常高峰)
latency[15*24:15*24+6] += 200  # 延迟增加200ms

df = pd.DataFrame({
    'ds': dates,
    'y': latency
})

# 训练
detector = ProphetAnomalyDetector(interval_width=0.95)
train_df = df.iloc[:20*24]  # 前20天训练
detector.fit(train_df)

# 检测
test_df = df.iloc[20*24:]  # 后10天检测
results = detector.detect(test_df)

# 输出异常
anomalies = results[results['is_anomaly']]
print(f"检测到 {len(anomalies)} 个异常点")
for _, row in anomalies.head().iterrows():
    print(f"时间: {row['timestamp']}, "
          f"实际值: {row['actual']:.1f}ms, "
          f"预测值: {row['predicted']:.1f}ms, "
          f"Z-Score: {row['zscore']:.2f}")
```

## 4. 根因分析 (RCA)

### 4.1 基于Trace的根因分析

```python
# 基于分布式追踪的根因分析
from dataclasses import dataclass
from typing import List, Dict, Optional
from collections import defaultdict

@dataclass
class Span:
    trace_id: str
    span_id: str
    parent_id: Optional[str]
    service_name: str
    operation_name: str
    start_time: int
    duration: int
    status: str
    attributes: Dict

class TraceRootCauseAnalyzer:
    """基于Trace的根因分析器"""

    def __init__(self):
        self.spans = []

    def add_spans(self, spans: List[Span]):
        """添加Span数据"""
        self.spans.extend(spans)

    def build_tree(self, trace_id: str) -> Dict:
        """构建调用树"""
        trace_spans = [s for s in self.spans if s.trace_id == trace_id]

        # 查找根Span
        root = None
        for span in trace_spans:
            if span.parent_id is None:
                root = span
                break

        if not root:
            return {}

        # 构建树
        def build_subtree(parent_span):
            children = [s for s in trace_spans if s.parent_id == parent_span.span_id]
            return {
                'span': parent_span,
                'children': [build_subtree(child) for child in children]
            }

        return build_subtree(root)

    def find_slow_spans(self, trace_id: str, threshold_ms: int = 100) -> List[Dict]:
        """查找慢Span"""
        trace_spans = [s for s in self.spans if s.trace_id == trace_id]

        slow_spans = []
        for span in trace_spans:
            if span.duration > threshold_ms * 1_000_000:  # 转换为纳秒
                slow_spans.append({
                    'service': span.service_name,
                    'operation': span.operation_name,
                    'duration_ms': span.duration / 1_000_000,
                    'status': span.status
                })

        return sorted(slow_spans, key=lambda x: x['duration_ms'], reverse=True)

    def find_error_spans(self, trace_id: str) -> List[Dict]:
        """查找错误Span"""
        trace_spans = [s for s in self.spans if s.trace_id == trace_id]

        error_spans = []
        for span in trace_spans:
            if span.status == 'ERROR':
                error_spans.append({
                    'service': span.service_name,
                    'operation': span.operation_name,
                    'duration_ms': span.duration / 1_000_000,
                    'error_type': span.attributes.get('error.type', 'unknown')
                })

        return error_spans

    def analyze_critical_path(self, trace_id: str) -> List[Dict]:
        """分析关键路径"""
        tree = self.build_tree(trace_id)

        if not tree:
            return []

        # 找出耗时最长的叶子到根的路径
        def find_longest_path(node, current_path=[]):
            current_path = current_path + [node['span']]

            if not node['children']:
                return current_path

            longest_path = current_path
            for child in node['children']:
                path = find_longest_path(child, current_path)
                total_duration = sum(s.duration for s in path)
                longest_total = sum(s.duration for s in longest_path)
                if total_duration > longest_total:
                    longest_path = path

            return longest_path

        critical_path = find_longest_path(tree)

        return [{
            'service': span.service_name,
            'operation': span.operation_name,
            'duration_ms': span.duration / 1_000_000,
            'percentage': span.duration / sum(s.duration for s in critical_path) * 100
        } for span in critical_path]

    def root_cause_analysis(self, trace_id: str) -> Dict:
        """执行根因分析"""
        tree = self.build_tree(trace_id)
        slow_spans = self.find_slow_spans(trace_id)
        error_spans = self.find_error_spans(trace_id)
        critical_path = self.analyze_critical_path(trace_id)

        # 推断根因
        root_causes = []

        if error_spans:
            # 按服务分组错误
            errors_by_service = defaultdict(list)
            for span in error_spans:
                errors_by_service[span['service']].append(span)

            # 找出错误最多的服务
            most_errors_service = max(errors_by_service.keys(),
                                     key=lambda s: len(errors_by_service[s]))
            root_causes.append({
                'type': 'error_propagation',
                'service': most_errors_service,
                'evidence': f"该服务产生 {len(errors_by_service[most_errors_service])} 个错误",
                'confidence': 'high'
            })

        if slow_spans:
            # 找出最慢的Span
            slowest = slow_spans[0]
            root_causes.append({
                'type': 'performance_degradation',
                'service': slowest['service'],
                'operation': slowest['operation'],
                'duration_ms': slowest['duration_ms'],
                'evidence': f"最慢操作耗时 {slowest['duration_ms']:.2f}ms",
                'confidence': 'medium'
            })

        return {
            'trace_id': trace_id,
            'root_causes': root_causes,
            'slow_spans': slow_spans[:5],
            'error_spans': error_spans,
            'critical_path': critical_path
        }

# 使用示例
analyzer = TraceRootCauseAnalyzer()
```

```python
# 智能告警抑制
from datetime import datetime, timedelta
from collections import deque

class SmartAlertSuppressor:
    """智能告警抑制器"""

    def __init__(self):
        self.alert_history = deque(maxlen=1000)
        self.suppression_rules = []

    def add_suppression_rule(self, rule):
        """添加抑制规则"""
        self.suppression_rules.append(rule)

    def should_suppress(self, alert):
        """判断是否抑制告警"""
        # 检查抑制规则
        for rule in self.suppression_rules:
            if rule(alert, self.alert_history):
                return True

        # 检查最近是否有相同告警
        recent_alerts = [
            a for a in self.alert_history
            if a['name'] == alert['name']
            and a['timestamp'] > datetime.now() - timedelta(minutes=5)
        ]

        if len(recent_alerts) >= 3:
            # 如果5分钟内已有3个相同告警，抑制
            return True

        return False

    def add_alert(self, alert):
        """记录告警"""
        self.alert_history.append({
            'name': alert['name'],
            'timestamp': datetime.now(),
            'severity': alert.get('severity', 'warning')
        })

# 抑制规则示例
def maintenance_window_rule(alert, history):
    """维护窗口期间抑制"""
    now = datetime.now()
    # 假设维护窗口是每周日凌晨2-4点
    if now.weekday() == 6 and 2 <= now.hour < 4:
        return True
    return False

def flapping_suppression_rule(alert, history):
    """抑制抖动告警"""
    # 检查最近10分钟内告警是否反复出现
    recent = [a for a in history
              if a['name'] == alert['name']
              and a['timestamp'] > datetime.now() - timedelta(minutes=10)]

    if len(recent) > 5:
        return True
    return False

# 使用
suppressor = SmartAlertSuppressor()
suppressor.add_suppression_rule(maintenance_window_rule)
suppressor.add_suppression_rule(flapping_suppression_rule)

# 测试告警
alert = {
    'name': 'HighErrorRate',
    'severity': 'critical',
    'value': 0.15
}

if not suppressor.should_suppress(alert):
    print("发送告警")
    suppressor.add_alert(alert)
else:
    print("告警被抑制")
```

```python
# 简单自愈系统
import asyncio

class SelfHealingSystem:
    """自愈系统"""

    def __init__(self):
        self.remediation_actions = {}
        self.execution_history = []

    def register_action(self, trigger, action):
        """注册修复动作"""
        self.remediation_actions[trigger] = action

    async def check_and_heal(self, metrics):
        """检查并执行修复"""
        for trigger, action in self.remediation_actions.items():
            if trigger(metrics):
                print(f"触发条件满足: {trigger.__name__}")

                # 检查最近是否执行过
                recent_executions = [
                    e for e in self.execution_history
                    if e['action'] == action.__name__
                    and e['timestamp'] > datetime.now() - timedelta(minutes=10)
                ]

                if recent_executions:
                    print(f"最近已执行过 {action.__name__}，跳过")
                    continue

                # 执行修复
                try:
                    result = await action(metrics)
                    self.execution_history.append({
                        'action': action.__name__,
                        'timestamp': datetime.now(),
                        'result': result
                    })
                    print(f"修复动作执行结果: {result}")
                except Exception as e:
                    print(f"修复动作失败: {e}")

# 定义触发条件和修复动作
def high_memory_trigger(metrics):
    return metrics.get('memory_usage', 0) > 90

async def restart_service(metrics):
    """重启服务"""
    service = metrics.get('service_name')
    print(f"正在重启服务: {service}")
    # 实际实现会调用K8s API或systemd
    await asyncio.sleep(2)
    return f"服务 {service} 已重启"

def high_error_rate_trigger(metrics):
    return metrics.get('error_rate', 0) > 0.1

async def scale_up_service(metrics):
    """扩容服务"""
    service = metrics.get('service_name')
    print(f"正在扩容服务: {service}")
    # 实际实现会调用K8s API进行HPA
    await asyncio.sleep(2)
    return f"服务 {service} 已扩容"

# 使用
healing_system = SelfHealingSystem()
healing_system.register_action(high_memory_trigger, restart_service)
healing_system.register_action(high_error_rate_trigger, scale_up_service)

# 模拟检查
async def main():
    metrics = {
        'service_name': 'order-service',
        'memory_usage': 95,
        'error_rate': 0.15
    }
    await healing_system.check_and_heal(metrics)

asyncio.run(main())
```

## 6. 总结

### 异常检测方法对比

| 方法 | 优点 | 缺点 | 适用场景 |
|------|------|------|----------|
| 统计阈值 | 简单、可解释 | 无法处理周期性 | 简单指标 |
| 时间序列分解 | 处理周期性 | 需要历史数据 | 业务指标 |
| 孤立森林 | 无需标签 | 高维效果一般 | 多维指标 |
| LSTM | 捕捉复杂模式 | 需要训练数据 | 大规模时序 |
| Prophet | 自动处理周期性 | 计算开销大 | 业务预测 |

### 根因分析最佳实践

```markdown
1. 数据收集
   - 确保Trace完整性
   - 收集相关Metrics和Logs
   - 记录系统变更事件

2. 分析流程
   - 从用户影响开始
   - 追踪调用链
   - 识别关键路径
   - 关联相关事件

3. 验证假设
   - 相关性不等于因果性
   - 多证据验证
   - 排除混淆因素
```

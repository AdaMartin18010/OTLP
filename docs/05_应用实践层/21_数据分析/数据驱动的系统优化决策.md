### 2.2 容量预测模型

```python
import pandas as pd
import numpy as np
from datetime import datetime, timedelta

class CapacityPlanner:
    """容量规划器"""

    def __init__(self, historical_data: pd.DataFrame):
        self.data = historical_data

    def linear_forecast(self, days_ahead: int = 90) -> pd.DataFrame:
        """线性增长预测"""
        df = self.data.copy()
        df['days'] = (df['timestamp'] - df['timestamp'].min()).dt.days

        # 计算增长率
        x = df['days'].values
        y = df['metric_value'].values

        # 简单线性回归
        slope = np.polyfit(x, y, 1)[0]
        intercept = y[-1]

        # 预测未来
        last_day = df['days'].max()
        future_days = np.arange(last_day + 1, last_day + days_ahead + 1)
        predictions = slope * future_days + intercept

        # 生成预测日期
        last_date = df['timestamp'].max()
        future_dates = [last_date + timedelta(days=int(d))
                       for d in range(1, days_ahead + 1)]

        return pd.DataFrame({
            'timestamp': future_dates,
            'predicted_value': predictions
        })

    def capacity_alert_timeline(
        self,
        forecast: pd.DataFrame,
        capacity_limit: float,
        threshold_percent: float = 0.8
    ) -> dict:
        """计算容量告警时间线"""
        threshold_value = capacity_limit * threshold_percent
        exceeding = forecast[forecast['predicted_value'] > threshold_value]

        if exceeding.empty:
            return {
                'status': 'OK',
                'message': f'预测期内不会达到{threshold_percent*100}%容量阈值'
            }

        alert_date = exceeding['timestamp'].min()
        days_until = (alert_date - datetime.now()).days

        return {
            'status': 'WARNING',
            'message': f'预计{days_until}天后达到{threshold_percent*100}%容量阈值',
            'alert_date': alert_date.strftime('%Y-%m-%d'),
            'days_until_alert': days_until
        }
```

### 3.2 性能瓶颈分析

```python
import time
import functools
from typing import Dict, List

class PerformanceProfiler:
    """性能分析器"""

    def __init__(self):
        self.metrics = {}

    def timeit(self, name: str):
        """装饰器：测量函数执行时间"""
        def decorator(func):
            @functools.wraps(func)
            def wrapper(*args, **kwargs):
                start = time.perf_counter()
                result = func(*args, **kwargs)
                elapsed = time.perf_counter() - start

                if name not in self.metrics:
                    self.metrics[name] = []
                self.metrics[name].append(elapsed)

                return result
            return wrapper
        return decorator

    def analyze_bottlenecks(self) -> List[Dict]:
        """分析性能瓶颈"""
        analysis = []
        total_time = sum(sum(times) for times in self.metrics.values())

        for name, times in self.metrics.items():
            total = sum(times)
            avg = sum(times) / len(times)

            analysis.append({
                'name': name,
                'call_count': len(times),
                'total_time': total,
                'avg_time_ms': avg * 1000,
                'percentage': (total / total_time * 100) if total_time > 0 else 0
            })

        return sorted(analysis, key=lambda x: x['percentage'], reverse=True)

    def report(self) -> str:
        """生成性能报告"""
        analysis = self.analyze_bottlenecks()
        lines = ["性能分析报告:", "-" * 50]
        for item in analysis:
            lines.append(f"{item['name']}: {item['percentage']:.1f}% "
                        f"({item['avg_time_ms']:.2f}ms avg)")
        return '\n'.join(lines)


# 使用示例
profiler = PerformanceProfiler()

@profiler.timeit("database_query")
def query_db():
    time.sleep(0.1)

@profiler.timeit("api_call")
def call_api():
    time.sleep(0.05)

# 模拟调用
for _ in range(10):
    query_db()
    call_api()

print(profiler.report())
```

### 4.2 成本优化决策工具

```python
from dataclasses import dataclass
from typing import List, Dict

@dataclass
class ResourceCost:
    """资源成本数据"""
    resource_type: str
    resource_id: str
    monthly_cost: float
    utilization_avg: float

class CostOptimizer:
    """云成本优化分析器"""

    UNDERUTILIZED_THRESHOLD = 0.30

    def analyze_resources(self, resources: List[ResourceCost]) -> Dict:
        """分析资源成本效率"""

        total_cost = sum(r.monthly_cost for r in resources)

        underutilized = [r for r in resources
                        if r.utilization_avg < self.UNDERUTILIZED_THRESHOLD]

        underutilized_cost = sum(r.monthly_cost for r in underutilized)
        potential_savings = underutilized_cost * 0.5

        return {
            'total_monthly_cost': total_cost,
            'underutilized_count': len(underutilized),
            'potential_savings': potential_savings,
            'savings_percentage': (potential_savings / total_cost * 100)
                                  if total_cost > 0 else 0
        }

    def right_size_recommendation(
        self,
        current_instance: str,
        cpu_avg: float,
        memory_avg: float
    ) -> Dict:
        """实例规格优化建议"""

        instance_specs = {
            'small': {'cpu': 2, 'memory': 4, 'cost_factor': 1},
            'medium': {'cpu': 4, 'memory': 8, 'cost_factor': 2},
            'large': {'cpu': 8, 'memory': 16, 'cost_factor': 4}
        }

        # 计算推荐规格 (预留30%缓冲)
        rec_cpu = cpu_avg * 1.3 * 100  # 转为百分比
        rec_memory = memory_avg * 1.3 * 100

        recommended = None
        for size, spec in sorted(instance_specs.items(),
                                 key=lambda x: x[1]['cost_factor']):
            if spec['cpu'] >= rec_cpu and spec['memory'] >= rec_memory:
                recommended = size
                break

        if recommended and recommended != current_instance:
            current = instance_specs[current_instance]['cost_factor']
            new = instance_specs[recommended]['cost_factor']

            return {
                'current': current_instance,
                'recommended': recommended,
                'saving_percent': (1 - new/current) * 100
            }

        return {'status': 'optimal', 'current': current_instance}


# 使用示例
resources = [
    ResourceCost('vm', 'web-01', 200, 0.15),
    ResourceCost('vm', 'api-01', 400, 0.85),
    ResourceCost('storage', 'data', 150, 0.10)
]

optimizer = CostOptimizer()
analysis = optimizer.analyze_resources(resources)

print(f"月度总成本: ${analysis['total_monthly_cost']}")
print(f"潜在节省: ${analysis['potential_savings']:.2f} "
      f"({analysis['savings_percentage']:.1f}%)")
```

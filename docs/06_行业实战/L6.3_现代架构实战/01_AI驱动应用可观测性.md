```python
from opentelemetry import trace, metrics
from opentelemetry.sdk.trace import TracerProvider
from opentelemetry.sdk.trace.export import BatchSpanProcessor
from opentelemetry.sdk.metrics import MeterProvider
from opentelemetry.sdk.metrics.export import PeriodicExportingMetricReader
from opentelemetry.exporter.otlp.proto.grpc.trace_exporter import OTLPSpanExporter
from opentelemetry.exporter.otlp.proto.grpc.metric_exporter import OTLPMetricExporter
from opentelemetry.sdk.resources import Resource, SERVICE_NAME, SERVICE_VERSION, DEPLOYMENT_ENVIRONMENT
from opentelemetry.instrumentation.fastapi import FastAPIInstrumentor
from opentelemetry.instrumentation.logging import LoggingInstrumentor
from prometheus_client import start_http_server
from app.core.config import settings

def init_telemetry(app):
    """初始化OpenTelemetry"""

    # 资源配置
    resource = Resource(attributes={
        SERVICE_NAME: settings.otel_service_name,
        SERVICE_VERSION: settings.app_version,
        DEPLOYMENT_ENVIRONMENT: "production",
        "service.namespace": "ai-platform",
        "host.name": os.uname().nodename,
    })

    # Tracer配置
    tracer_provider = TracerProvider(resource=resource)
    otlp_trace_exporter = OTLPSpanExporter(endpoint=settings.otel_endpoint)
    tracer_provider.add_span_processor(BatchSpanProcessor(otlp_trace_exporter))
    trace.set_tracer_provider(tracer_provider)

    # Meter配置
    metric_reader = PeriodicExportingMetricReader(OTLPMetricExporter(endpoint=settings.otel_endpoint))
    meter_provider = MeterProvider(resource=resource, metric_readers=[metric_reader])
    metrics.set_meter_provider(meter_provider)

    # 自动仪器化
    FastAPIInstrumentor.instrument_app(app)
    LoggingInstrumentor().instrument()

    # Prometheus指标端点
    start_http_server(settings.otel_prometheus_port)

    return trace.get_tracer(__name__), metrics.get_meter(__name__)

# 创建全局tracer和meter
```

## 3. LLM服务实现

### 3.1 带仪器化的LLM客户端

```python
# app/services/llm_service.py
import time
import json
from typing import List, Dict, Any, Optional
from openai import OpenAI, AsyncOpenAI
from opentelemetry.trace import Status, StatusCode
from app.core.telemetry import tracer, gen_ai_token_counter, gen_ai_latency_histogram, gen_ai_cost_counter
from app.core.config import settings

# 模型定价 (USD per 1M tokens)
MODEL_PRICING = {
    "gpt-4o": {"input": 2.50, "output": 10.00},
    "gpt-4o-mini": {"input": 0.15, "output": 0.60},
    "text-embedding-3-small": {"input": 0.02, "output": 0.0},
}

class LLMService:
    def __init__(self):
        self.client = AsyncOpenAI(api_key=settings.openai_api_key)
        self.model = settings.openai_model

    def _calculate_cost(self, model: str, input_tokens: int, output_tokens: int) -> float:
        """计算API调用成本"""
        pricing = MODEL_PRICING.get(model, MODEL_PRICING["gpt-4o"])
        input_cost = (input_tokens / 1_000_000) * pricing["input"]
        output_cost = (output_tokens / 1_000_000) * pricing["output"]
        return round(input_cost + output_cost, 6)

    @tracer.start_as_current_span("llm.chat")
    async def chat(
        self,
        messages: List[Dict[str, str]],
        session_id: Optional[str] = None,
        user_id: Optional[str] = None,
        **kwargs
    ) -> Dict[str, Any]:
        """对话生成"""

        span = trace.get_current_span()

        # GenAI语义约定属性
        span.set_attribute("gen_ai.system", "openai")
        span.set_attribute("gen_ai.operation.name", "chat")
        span.set_attribute("gen_ai.request.model", self.model)

        if session_id:
            span.set_attribute("session.id", session_id)
        if user_id:
            span.set_attribute("user.id", user_id)

        span.set_attribute("gen_ai.request.temperature", kwargs.get("temperature", 0.7))
        span.set_attribute("gen_ai.request.max_tokens", kwargs.get("max_tokens", 2000))
        span.set_attribute("gen_ai.request.message_count", len(messages))

        start_time = time.time()

        try:
            response = await self.client.chat.completions.create(
                model=self.model,
                messages=messages,
                **kwargs
            )

            latency_ms = (time.time() - start_time) * 1000

            # 记录响应
            span.set_attribute("gen_ai.response.model", response.model)
            span.set_attribute("gen_ai.response.id", response.id)
            span.set_attribute("gen_ai.response.finish_reason",
                             response.choices[0].finish_reason)

            # Token使用
            input_tokens = response.usage.prompt_tokens
            output_tokens = response.usage.completion_tokens

            span.set_attribute("gen_ai.usage.input_tokens", input_tokens)
            span.set_attribute("gen_ai.usage.output_tokens", output_tokens)
            span.set_attribute("gen_ai.usage.total_tokens",
                             response.usage.total_tokens)

            # 成本和延迟
            cost = self._calculate_cost(self.model, input_tokens, output_tokens)
            span.set_attribute("gen_ai.estimated_cost_usd", cost)
            span.set_attribute("gen_ai.latency_ms", latency_ms)

            # 记录指标
            gen_ai_token_counter.add(response.usage.total_tokens, {
                "model": self.model,
                "operation": "chat"
            })
            gen_ai_latency_histogram.record(latency_ms, {"model": self.model})
            gen_ai_cost_counter.add(cost, {"model": self.model})

            # 警告检查
            if response.choices[0].finish_reason == "length":
                span.set_attribute("gen_ai.warning", "response_truncated")

            return {
                "content": response.choices[0].message.content,
                "model": response.model,
                "usage": {
                    "input_tokens": input_tokens,
                    "output_tokens": output_tokens,
                    "total_tokens": response.usage.total_tokens
                },
                "cost_usd": cost,
                "latency_ms": latency_ms
            }

        except Exception as e:
            span.set_attribute("error.type", type(e).__name__)
            span.set_attribute("error.message", str(e))
            span.set_status(Status(StatusCode.ERROR, str(e)))
            raise

    @tracer.start_as_current_span("llm.embedding")
    async def create_embedding(self, text: str) -> List[float]:
        """创建嵌入向量"""

        span = trace.get_current_span()
        span.set_attribute("gen_ai.system", "openai")
        span.set_attribute("gen_ai.operation.name", "embeddings")
        span.set_attribute("gen_ai.request.model", settings.embedding_model)

        response = await self.client.embeddings.create(
            model=settings.embedding_model,
            input=text
        )

        span.set_attribute("gen_ai.usage.total_tokens", response.usage.total_tokens)
        span.set_attribute("gen_ai.embedding.dimensions", len(response.data[0].embedding))

        return response.data[0].embedding
```

## 4. RAG服务实现

```python
# app/services/rag_service.py
import chromadb
from chromadb.utils import embedding_functions
from typing import List, Dict
from opentelemetry import trace
from app.core.telemetry import tracer

class RAGService:
    def __init__(self):
        self.client = chromadb.Client()
        self.embedding_func = embedding_functions.OpenAIEmbeddingFunction(
            api_key=settings.openai_api_key,
            model_name=settings.embedding_model
        )

        # 创建或获取集合
        self.collection = self.client.get_or_create_collection(
            name="knowledge_base",
            embedding_function=self.embedding_func
        )

        # 初始化知识库
        self._init_knowledge()

    @tracer.start_as_current_span("rag.retrieve")
    async def retrieve(
        self,
        query: str,
        top_k: int = 3,
        threshold: float = 0.7
    ) -> List[Dict]:
        """检索相关知识"""

        span = trace.get_current_span()
        span.set_attribute("rag.query", query[:100])
        span.set_attribute("rag.top_k", top_k)
        span.set_attribute("rag.threshold", threshold)

        results = self.collection.query(
            query_texts=[query],
            n_results=top_k
        )

        documents = []
        for i, (doc, metadata, distance) in enumerate(zip(
            results['documents'][0],
            results['metadatas'][0],
            results['distances'][0]
        )):
            # 转换距离为相似度
            similarity = 1 - distance
            if similarity >= threshold:
                documents.append({
                    "content": doc,
                    "metadata": metadata,
                    "similarity": similarity
                })

        span.set_attribute("rag.results_count", len(documents))
        span.set_attribute("rag.avg_similarity",
                         sum(d["similarity"] for d in documents) / len(documents) if documents else 0)

        return documents
```

## 5. API实现

```python
# app/api/chat.py
from fastapi import APIRouter, HTTPException, Depends
from pydantic import BaseModel
from typing import Optional, List
from app.services.llm_service import LLMService
from app.services.rag_service import RAGService
from app.core.telemetry import tracer, request_counter

router = APIRouter()
llm_service = LLMService()
rag_service = RAGService()

class ChatRequest(BaseModel):
    message: str
    session_id: Optional[str] = None
    user_id: Optional[str] = None
    use_rag: bool = True

class ChatResponse(BaseModel):
    response: str
    sources: Optional[List[dict]] = None
    model: str
    usage: dict
    cost_usd: float
    latency_ms: float

@router.post("/chat", response_model=ChatResponse)
@tracer.start_as_current_span("api.chat")
async def chat_endpoint(request: ChatRequest):
    """对话接口"""

    span = trace.get_current_span()
    span.set_attribute("http.route", "/chat")
    span.set_attribute("user.id", request.user_id or "anonymous")

    request_counter.add(1, {"endpoint": "/chat", "method": "POST"})

    try:
        # 检索相关知识
        context = ""
        sources = None
        if request.use_rag:
            with tracer.start_as_current_span("rag.retrieve") as rag_span:
                rag_span.set_attribute("session.id", request.session_id)
                documents = await rag_service.retrieve(request.message)

                if documents:
                    context = "\n\n".join([f"[{i+1}] {doc['content']}"
                                         for i, doc in enumerate(documents)])
                    sources = documents

                    rag_span.set_attribute("rag.context_length", len(context))

        # 构建消息
        system_prompt = """你是智能客服助手。请基于以下知识回答用户问题：

{context}

要求：
1. 回答简洁专业
2. 引用知识来源[n]
3. 不确定时建议转人工""".format(context=context or "无相关知识")

        messages = [
            {"role": "system", "content": system_prompt},
            {"role": "user", "content": request.message}
        ]

        # 调用LLM
        result = await llm_service.chat(
            messages=messages,
            session_id=request.session_id,
            user_id=request.user_id,
            temperature=0.7,
            max_tokens=1500
        )

        return ChatResponse(
            response=result["content"],
            sources=sources,
            model=result["model"],
            usage=result["usage"],
            cost_usd=result["cost_usd"],
            latency_ms=result["latency_ms"]
        )

    except Exception as e:
        span.set_attribute("error.type", type(e).__name__)
        span.set_attribute("error.message", str(e))
        raise HTTPException(status_code=500, detail=str(e))
```

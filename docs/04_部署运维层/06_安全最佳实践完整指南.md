## 2. 传输层安全

### 2.1 mTLS完整配置

```yaml
# 服务端配置 (Collector)
receivers:
  otlp:
    protocols:
      grpc:
        endpoint: 0.0.0.0:4317
        tls:
          # 服务端证书
          cert_file: /etc/otel/certs/server.crt
          key_file: /etc/otel/certs/server.key
          
          # 客户端CA证书 (启用mTLS)
          client_ca_file: /etc/otel/certs/ca.crt
          
          # 客户端认证要求
          client_auth_type: "require_and_verify_client_cert"
          # 可选值:
          # - "no_client_auth": 不要求客户端证书
          # - "request_client_auth": 请求但不验证
          # - "require_any_client_auth": 要求任意证书
          # - "verify_client_cert_if_given": 如有则验证
          # - "require_and_verify_client_cert": 要求并验证 (推荐)
          
          # TLS版本强制
          min_version: "1.3"
          # 允许的密码套件
          cipher_suites: [
            "TLS_AES_256_GCM_SHA384",
            "TLS_CHACHA20_POLY1305_SHA256",
            "TLS_AES_128_GCM_SHA256"
          ]
          
          # 重协商限制
          reload_interval: 1h  # 证书热重载间隔
      
      http:
        endpoint: 0.0.0.0:4318
        tls:
          cert_file: /etc/otel/certs/server.crt
          key_file: /etc/otel/certs/server.key
          client_ca_file: /etc/otel/certs/ca.crt
          client_auth_type: "require_and_verify_client_cert"
          min_version: "1.3"

# 客户端配置 (SDK或Collector作为客户端)
exporters:
  otlp:
    endpoint: backend-collector:4317
    tls:
      # 客户端证书
      cert_file: /etc/otel/certs/client.crt
      key_file: /etc/otel/certs/client.key
      
      # 服务端CA证书
      ca_file: /etc/otel/certs/ca.crt
      
      # 禁用不安全选项
      insecure: false
      insecure_skip_verify: false
      
      # 服务端名称验证 (防止中间人攻击)
      server_name_override: "otel-collector.backend.svc.cluster.local"
```

### 2.3 TLS版本与密码套件配置

```yaml
# 安全TLS配置
tls_security_hardening:
  # 禁止不安全的TLS版本
  disallowed_versions:
    - "1.0"
    - "1.1"
  
  # 推荐的TLS配置
  recommended_config:
    min_version: "1.3"  # 首选TLS 1.3
    fallback_version: "1.2"  # 向后兼容时使用1.2
    
    # TLS 1.3密码套件 (无需配置，自动协商)
    # TLS 1.2密码套件配置
    tls12_cipher_suites:
      recommended:
        - "TLS_ECDHE_RSA_WITH_AES_256_GCM_SHA384"
        - "TLS_ECDHE_RSA_WITH_CHACHA20_POLY1305_SHA256"
        - "TLS_ECDHE_RSA_WITH_AES_128_GCM_SHA256"
      
      disallowed:
        - "TLS_RSA_WITH_3DES_EDE_CBC_SHA"  # 弱密码
        - "TLS_RSA_WITH_RC4_128_SHA"       # 已破解
        - "TLS_RSA_WITH_AES_128_CBC_SHA"   # 无前向保密
  
  # 合规性检查脚本
  compliance_check: |
    #!/bin/bash
    echo "检查TLS配置合规性..."
    
    # 检查最低TLS版本
    if openssl s_client -connect otel-collector:4317 -tls1_3 2>/dev/null | grep -q "Protocol.*TLSv1.3"; then
        echo "✓ 支持TLS 1.3"
    else
        echo "✗ 不支持TLS 1.3"
    fi
    
    # 检查弱密码
    weak_ciphers=$(openssl s_client -connect otel-collector:4317 -cipher 'ALL:eNULL' 2>/dev/null | grep "Cipher" | grep -i "null\|export\|des\|md5\|sha1")
    if [ -z "$weak_ciphers" ]; then
        echo "✓ 无弱密码"
    else
        echo "✗ 发现弱密码: $weak_ciphers"
    fi
```

### 3.2 传输中数据加密

```yaml
# 传输加密配置
transport_encryption:
  # 强制HTTPS
  http_enforcement:
    http_forwarder:
      # 重定向HTTP到HTTPS
      redirect_http_to_https: true
      
      # HSTS配置
      hsts:
        enabled: true
        max_age: 31536000  # 1年
        include_subdomains: true
        preload: true
  
  # gRPC加密
  grpc_encryption:
    otlp:
      tls:
        enabled: true
        # 强制加密，禁止明文传输
        allow_insecure: false
  
  # 压缩与加密
  compression_security:
    # 优先使用支持加密的压缩算法
    preferred_algorithms:
      - zstd  # 推荐：高压缩率，支持流式
      - gzip  # 兼容性最好
    
    # 禁用不安全的压缩
    disabled_algorithms:
      - none  # 禁止明文传输
  
  # 密钥交换前向保密
  forward_secrecy:
    enabled: true
    ephemeral_key_exchange:
      - ECDHE  # 椭圆曲线Diffie-Hellman
      - DHE    # 传统Diffie-Hellman (向后兼容)
```

## 4. 认证与授权

### 4.1 多因素认证配置

```yaml
# 认证配置
authentication:
  # Basic Auth (适用于内部服务)
  basic_auth:
    extensions:
      basicauth:
        client_auth:
          username: ${OTEL_USERNAME}
          password: ${OTEL_PASSWORD}
    
    receivers:
      otlp:
        protocols:
          http:
            auth:
              authenticator: basicauth
  
  # Bearer Token (适用于API访问)
  bearer_token:
    extensions:
      bearertokenauth:
        token: ${OTEL_API_TOKEN}
        # Token刷新配置
        refresh_interval: 1h
    
    exporters:
      otlp:
        endpoint: backend:4317
        auth:
          authenticator: bearertokenauth
  
  # OAuth2 (适用于第三方集成)
  oauth2:
    extensions:
      oauth2client:
        client_id: ${CLIENT_ID}
        client_secret: ${CLIENT_SECRET}
        token_url: https://auth.example.com/oauth/token
        scopes: ["otel:write", "otel:read"]
        
        # Token缓存
        token_cache:
          enabled: true
          ttl: 3500s  # 略短于token有效期
        
        # 自动刷新
        automatic_refresh: true
  
  # OIDC (适用于企业SSO)
  oidc:
    extensions:
      oidc:
        issuer_url: https://auth.example.com
        audience: otel-collector
        
        # 声明映射
        username_claim: email
        groups_claim: groups
        
        # 本地JWKS缓存
        local_jwks:
          path: /etc/otel/jwks.json
          refresh_interval: 1h
        
        # 允许的域名
        allowed_domains:
          - example.com
          - subsidiaries.example.com
  
  # mTLS (最安全的选项)
  mtls_auth:
    receivers:
      otlp:
        protocols:
          grpc:
            tls:
              client_auth_type: "require_and_verify_client_cert"
              client_ca_file: /etc/otel/certs/ca.crt
              
            # 基于证书的身份识别
            client_identity:
              extract_from: subject_common_name
              mapping:
                "service-*.example.com": service_account
                "user-*.example.com": user_account
```

## 5. 访问控制

### 5.1 网络隔离

```yaml
# 网络隔离配置
network_isolation:
  # Kubernetes NetworkPolicy
  network_policy:
    apiVersion: networking.k8s.io/v1
    kind: NetworkPolicy
    metadata:
      name: otel-collector-network-policy
      namespace: observability
    spec:
      podSelector:
        matchLabels:
          app: otel-collector
      policyTypes:
        - Ingress
        - Egress
      
      # 入站规则
      ingress:
        # 只允许特定命名空间的应用访问
        - from:
            - namespaceSelector:
                matchLabels:
                  otel-ingest: "true"
          ports:
            - protocol: TCP
              port: 4317  # OTLP gRPC
            - protocol: TCP
              port: 4318  # OTLP HTTP
        
        # 允许监控系统访问自监控端点
        - from:
            - namespaceSelector:
                matchLabels:
                  name: monitoring
          ports:
            - protocol: TCP
              port: 8888   # Metrics
            - protocol: TCP
              port: 13133  # Health
      
      # 出站规则
      egress:
        # 只允许访问后端存储
        - to:
            - namespaceSelector:
                matchLabels:
                  name: observability-backend
          ports:
            - protocol: TCP
              port: 4317
        
        # DNS查询
        - to:
            - namespaceSelector: {}
          ports:
            - protocol: UDP
              port: 53
  
  # 防火墙规则 (非K8s环境)
  firewall_rules:
    iptables:
      # 允许的应用服务器IP段
      - chain: INPUT
        source: 10.0.0.0/8
        destination_port: 4317
        action: ACCEPT
      
      # 拒绝其他来源
      - chain: INPUT
        destination_port: 4317
        action: DROP
      
      # 管理接口仅允许堡垒机
      - chain: INPUT
        source: 10.0.100.0/24
        destination_port: 13133
        action: ACCEPT
      
      - chain: INPUT
        destination_port: 13133
        action: DROP
```

# 数据脱敏处理器配置
processors:
  # 属性级别脱敏
  attributes/sanitize:
    actions:
      # 删除敏感头
      - key: http.request.header.authorization
        action: delete
      
      - key: http.request.header.cookie
        action: delete
      
      - key: http.request.header.x-api-key
        action: delete
      
      # 哈希处理 (保留统计价值)
      - key: user.email
        action: hash
        algorithm: sha256
        salt: ${HASH_SALT}  # 防止彩虹表攻击
      
      - key: user.id
        action: hash
        algorithm: sha256
      
      # 掩码处理
      - key: user.phone
        action: mask
        pattern: "(\\d{3})\\d{4}(\\d{4})"
        mask: "$1****$2"
      
      - key: credit_card.number
        action: mask
        pattern: "\\d{12}(\\d{4})"
        mask: "************$1"
      
      # 截断处理
      - key: http.request.body
        action: truncate
        max_length: 1000
        suffix: "...[truncated]"
      
      # 部分加密
      - key: db.statement
        action: encrypt_field
        fields:
          - pattern: "VALUES\\s*\\(([^)]+)\\)"
            encrypt: true
      
      # 正则替换
      - key: http.url
        action: regex_replace
        pattern: "token=[^&]*"
        replacement: "token=***"
  
  # 专门的PII脱敏处理器
  pii_redaction:
    detectors:
      - email
      - phone
      - ssn
      - credit_card
    
    actions:
      email:
        method: hash
        preserve_domain: true  # 保留域名用于分析
      
      phone:
        method: mask
        mask_pattern: "***-***-####"
      
      ssn:
        method: delete
      
      credit_card:
        method: mask
        mask_pattern: "****-****-****-####"
  
  # 条件脱敏
  conditional_sanitization:
    rules:
      # 生产环境完全脱敏
      - condition: 'attributes["deployment.environment"] == "production"'
        action: strict_redaction
      
      # 测试环境部分脱敏
      - condition: 'attributes["deployment.environment"] == "staging"'
        action: partial_redaction
      
      # 开发环境保持原样
      - condition: 'attributes["deployment.environment"] == "development"'
        action: none

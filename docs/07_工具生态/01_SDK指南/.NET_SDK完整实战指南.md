## 2. 基础配置

### 2.1 Program.cs 配置

```csharp
// Program.cs

using OpenTelemetry.Logs;
using OpenTelemetry.Metrics;
using OpenTelemetry.Resources;
using OpenTelemetry.Trace;
using MyApp.Web.Services;
using MyApp.Web.Telemetry;

var builder = WebApplication.CreateBuilder(args);

// 配置OpenTelemetry
ConfigureOpenTelemetry(builder);

// 添加服务
builder.Services.AddControllers();
builder.Services.AddEndpointsApiExplorer();
builder.Services.AddSwaggerGen();

// 注册业务服务
builder.Services.AddScoped<IOrderService, OrderService>();

var app = builder.Build();

// 配置HTTP请求管道
if (app.Environment.IsDevelopment())
{
    app.UseSwagger();
    app.UseSwaggerUI();
}

app.UseHttpsRedirection();
app.UseAuthorization();
app.MapControllers();

app.Run();

// OpenTelemetry配置方法
void ConfigureOpenTelemetry(WebApplicationBuilder builder)
{
    var serviceName = builder.Configuration["OTEL_SERVICE_NAME"] ?? "my-dotnet-service";
    var serviceVersion = builder.Configuration["OTEL_SERVICE_VERSION"] ?? "1.0.0";
    var otlpEndpoint = builder.Configuration["OTEL_EXPORTER_OTLP_ENDPOINT"] ?? "http://localhost:4317";

    // 配置资源
    var resourceBuilder = ResourceBuilder.CreateDefault()
        .AddService(serviceName, serviceVersion: serviceVersion)
        .AddAttributes(new Dictionary<string, object>
        {
            ["deployment.environment"] = builder.Environment.EnvironmentName,
            ["host.name"] = Environment.MachineName,
        })
        .AddContainerDetector()
        .AddHostDetector()
        .AddProcessRuntimeDetector();

    // 配置Tracing
    builder.Services.AddOpenTelemetry()
        .WithTracing(tracing =>
        {
            tracing
                .SetResourceBuilder(resourceBuilder)
                .AddAspNetCoreInstrumentation(options =>
                {
                    // 过滤健康检查
                    options.Filter = (httpContext) =>
                    {
                        return !httpContext.Request.Path.StartsWithSegments("/health");
                    };

                    // 记录请求头
                    options.RecordException = true;
                    options.EnrichWithHttpRequest = (activity, request) =>
                    {
                        activity.SetTag("http.request.body.size", request.ContentLength);
                        activity.SetTag("http.request.content_type", request.ContentType);
                    };
                    options.EnrichWithHttpResponse = (activity, response) =>
                    {
                        activity.SetTag("http.response.body.size", response.ContentLength);
                    };
                    options.EnrichWithException = (activity, exception) =>
                    {
                        activity.SetTag("exception.stacktrace", exception.StackTrace);
                    };
                })
                .AddHttpClientInstrumentation(options =>
                {
                    options.RecordException = true;
                    options.EnrichWithHttpRequestMessage = (activity, request) =>
                    {
                        activity.SetTag("http.request.headers.count", request.Headers.Count());
                    };
                })
                .AddSqlClientInstrumentation(options =>
                {
                    options.SetDbStatementForText = true;
                    options.RecordException = true;
                })
                .AddSource("MyApp.OrderService")
                .AddSource("MyApp.PaymentService")
                .AddOtlpExporter(options =>
                {
                    options.Endpoint = new Uri(otlpEndpoint);
                    options.TimeoutMilliseconds = 10000;

                    // 使用gRPC协议
                    options.Protocol = OpenTelemetry.Exporter.OtlpExportProtocol.Grpc;
                });

            // 采样配置
            if (builder.Environment.IsProduction())
            {
                tracing.SetSampler(new ParentBasedSampler(
                    new TraceIdRatioBasedSampler(0.1))); // 10%采样
            }
        })
        // 配置Metrics
        .WithMetrics(metrics =>
        {
            metrics
                .SetResourceBuilder(resourceBuilder)
                .AddAspNetCoreInstrumentation()
                .AddHttpClientInstrumentation()
                .AddRuntimeInstrumentation()
                .AddMeter("MyApp.OrderService")
                .AddMeter("MyApp.Metrics")
                .AddPrometheusExporter()
                .AddOtlpExporter(options =>
                {
                    options.Endpoint = new Uri(otlpEndpoint);
                    options.PeriodicExportingMetricReaderOptions = new PeriodicExportingMetricReaderOptions
                    {
                        ExportIntervalMilliseconds = 60000,
                    };
                });
        })
        // 配置Logs
        .WithLogging(logging =>
        {
            logging
                .SetResourceBuilder(resourceBuilder)
                .AddOtlpExporter(options =>
                {
                    options.Endpoint = new Uri(otlpEndpoint);
                });
        });

    // 配置日志
    builder.Logging.AddOpenTelemetry(logging =>
    {
        logging.IncludeFormattedMessage = true;
        logging.IncludeScopes = true;
        logging.ParseStateValues = true;
    });
}
```

### 2.3 模块化配置方式

```csharp
// Telemetry/OpenTelemetryConfig.cs

using OpenTelemetry.Logs;
using OpenTelemetry.Metrics;
using OpenTelemetry.Resources;
using OpenTelemetry.Trace;

namespace MyApp.Web.Telemetry;

public static class OpenTelemetryConfig
{
    public static IServiceCollection AddCustomOpenTelemetry(
        this IServiceCollection services,
        IConfiguration configuration,
        IWebHostEnvironment environment)
    {
        var serviceName = configuration["OTEL_SERVICE_NAME"] ?? "my-service";
        var serviceVersion = configuration["OTEL_SERVICE_VERSION"] ?? "1.0.0";
        var otlpEndpoint = configuration["OTEL_EXPORTER_OTLP_ENDPOINT"] ?? "http://localhost:4317";

        var resourceBuilder = ResourceBuilder.CreateDefault()
            .AddService(serviceName, serviceVersion: serviceVersion)
            .AddContainerDetector()
            .AddHostDetector();

        services.AddOpenTelemetry()
            .WithTracing(tracing => ConfigureTracing(tracing, resourceBuilder, otlpEndpoint, environment))
            .WithMetrics(metrics => ConfigureMetrics(metrics, resourceBuilder, otlpEndpoint));

        return services;
    }

    private static TracerProviderBuilder ConfigureTracing(
        TracerProviderBuilder tracing,
        ResourceBuilder resourceBuilder,
        string otlpEndpoint,
        IWebHostEnvironment environment)
    {
        tracing
            .SetResourceBuilder(resourceBuilder)
            .AddAspNetCoreInstrumentation()
            .AddHttpClientInstrumentation()
            .AddSource("MyApp.*")
            .AddOtlpExporter(options =>
            {
                options.Endpoint = new Uri(otlpEndpoint);
            });

        if (environment.IsProduction())
        {
            tracing.SetSampler(new TraceIdRatioBasedSampler(0.1));
        }

        return tracing;
    }

    private static MeterProviderBuilder ConfigureMetrics(
        MeterProviderBuilder metrics,
        ResourceBuilder resourceBuilder,
        string otlpEndpoint)
    {
        metrics
            .SetResourceBuilder(resourceBuilder)
            .AddAspNetCoreInstrumentation()
            .AddRuntimeInstrumentation()
            .AddMeter("MyApp.*")
            .AddOtlpExporter(options =>
            {
                options.Endpoint = new Uri(otlpEndpoint);
            });

        return metrics;
    }
}
```

## 3. 核心使用示例

### 3.1 创建Activity（Span）

```csharp
// Services/OrderService.cs

using System.Diagnostics;
using OpenTelemetry.Trace;
using MyApp.Web.Telemetry;

namespace MyApp.Web.Services;

public interface IOrderService
{
    Task<Order> CreateOrderAsync(string userId, decimal amount);
    Task<Order?> GetOrderAsync(string orderId);
    Task<List<Order>> ProcessBulkOrdersAsync(List<string> orderIds);
}

public class OrderService : IOrderService
{
    private static readonly ActivitySource ActivitySource = new("MyApp.OrderService");
    private readonly ILogger<OrderService> _logger;
    private readonly IPaymentService _paymentService;

    public OrderService(ILogger<OrderService> logger, IPaymentService paymentService)
    {
        _logger = logger;
        _paymentService = paymentService;
    }

    public async Task<Order> CreateOrderAsync(string userId, decimal amount)
    {
        // 方式1: 使用 ActivitySource 创建 Activity
        using var activity = ActivitySource.StartActivity("OrderService.CreateOrder", ActivityKind.Internal);

        try
        {
            // 设置标签
            activity?.SetTag("order.user_id", userId);
            activity?.SetTag("order.amount", amount);

            // 记录事件
            activity?.AddEvent(new ActivityEvent("Order validation started"));
            await ValidateOrderAsync(userId, amount);
            activity?.AddEvent(new ActivityEvent("Order validation completed"));

            // 生成订单ID
            var orderId = Guid.NewGuid().ToString();
            activity?.SetTag("order.id", orderId);

            // 处理支付
            activity?.AddEvent(new ActivityEvent("Payment processing started"));
            var paymentSuccess = await _paymentService.ProcessPaymentAsync(orderId, amount);

            if (!paymentSuccess)
            {
                throw new InvalidOperationException("Payment processing failed");
            }

            activity?.AddEvent(new ActivityEvent("Payment completed", tags: new ActivityTagsCollection
            {
                { "payment.transaction_id", $"txn_{orderId}" },
                { "payment.amount", amount }
            }));

            // 设置成功状态
            activity?.SetStatus(ActivityStatusCode.Ok);

            return new Order
            {
                OrderId = orderId,
                UserId = userId,
                Amount = amount,
                Status = "Created"
            };
        }
        catch (Exception ex)
        {
            // 记录异常
            activity?.RecordException(ex);
            activity?.SetStatus(ActivityStatusCode.Error, ex.Message);
            throw;
        }
    }

    private async Task ValidateOrderAsync(string userId, decimal amount)
    {
        // 方式2: 嵌套 Activity
        using var activity = ActivitySource.StartActivity("OrderService.ValidateOrder");

        activity?.SetTag("validation.user_id", userId);

        if (amount <= 0)
        {
            throw new ArgumentException("Amount must be greater than 0");
        }

        activity?.SetTag("validation.passed", true);
        await Task.Delay(10); // 模拟处理
    }

    public async Task<Order?> GetOrderAsync(string orderId)
    {
        using var activity = ActivitySource.StartActivity("OrderService.GetOrder");
        activity?.SetTag("order.id", orderId);

        try
        {
            // 模拟数据库查询
            await Task.Delay(20);

            var order = new Order
            {
                OrderId = orderId,
                UserId = "user123",
                Amount = 99.99m,
                Status = "Completed"
            };

            activity?.SetTag("order.found", true);
            activity?.SetTag("order.status", order.Status);
            activity?.SetStatus(ActivityStatusCode.Ok);

            return order;
        }
        catch (Exception ex)
        {
            activity?.RecordException(ex);
            activity?.SetStatus(ActivityStatusCode.Error, ex.Message);
            throw;
        }
    }

    public async Task<List<Order>> ProcessBulkOrdersAsync(List<string> orderIds)
    {
        // 方式3: 父 Activity 和多个子 Activity
        using var parentActivity = ActivitySource.StartActivity("OrderService.ProcessBulkOrders");
        parentActivity?.SetTag("order.count", orderIds.Count);

        var results = new List<Order>();

        foreach (var orderId in orderIds)
        {
            // 创建子 Activity
            using var childActivity = ActivitySource.StartActivity(
                "OrderService.ProcessSingleOrder",
                ActivityKind.Internal,
                parentActivity?.Context ?? default);

            childActivity?.SetTag("order.id", orderId);

            try
            {
                var order = await GetOrderAsync(orderId);
                if (order != null)
                {
                    childActivity?.SetTag("order.status", order.Status);
                    childActivity?.SetStatus(ActivityStatusCode.Ok);
                    results.Add(order);
                }
            }
            catch (Exception ex)
            {
                childActivity?.RecordException(ex);
                childActivity?.SetStatus(ActivityStatusCode.Error, ex.Message);
            }
        }

        parentActivity?.SetTag("order.processed_count", results.Count);
        parentActivity?.SetStatus(ActivityStatusCode.Ok);

        return results;
    }
}

// 订单模型
public class Order
{
    public string OrderId { get; set; } = string.Empty;
    public string UserId { get; set; } = string.Empty;
    public decimal Amount { get; set; }
    public string Status { get; set; } = string.Empty;
}
```

### 3.2 记录指标

```csharp
// Telemetry/MetricsConfig.cs

using System.Diagnostics.Metrics;

namespace MyApp.Web.Telemetry;

public class OrderMetrics
{
    private readonly Meter _meter;
    private readonly Counter<long> _orderCounter;
    private readonly Counter<long> _orderErrorCounter;
    private readonly Histogram<double> _orderValueHistogram;
    private readonly Histogram<double> _orderProcessingTime;
    private readonly ObservableGauge<long> _queueSizeGauge;

    private long _queueSize = 0;

    public OrderMetrics(IMeterFactory meterFactory)
    {
        _meter = meterFactory.Create("MyApp.OrderService");

        // 计数器
        _orderCounter = _meter.CreateCounter<long>(
            "orders.total",
            description: "Total number of orders",
            unit: "1");

        _orderErrorCounter = _meter.CreateCounter<long>(
            "orders.errors",
            description: "Total number of order errors",
            unit: "1");

        // 直方图
        _orderValueHistogram = _meter.CreateHistogram<double>(
            "order.value",
            description: "Order value distribution",
            unit: "USD");

        _orderProcessingTime = _meter.CreateHistogram<double>(
            "order.processing.duration",
            description: "Order processing time",
            unit: "ms");

        // Observable Gauge
        _queueSizeGauge = _meter.CreateObservableGauge(
            "queue.size",
            () => new Measurement<long>(_queueSize),
            description: "Current queue size",
            unit: "1");
    }

    public void RecordOrderCreated(decimal amount, string status = "success")
    {
        _orderCounter.Add(1,
            new KeyValuePair<string, object?>("status", status),
            new KeyValuePair<string, object?>("service", "order-service"));

        _orderValueHistogram.Record((double)amount,
            new KeyValuePair<string, object?>("currency", "USD"));
    }

    public void RecordOrderError(string errorType)
    {
        _orderErrorCounter.Add(1,
            new KeyValuePair<string, object?>("error.type", errorType));
    }

    public void RecordProcessingTime(double durationMs, string method)
    {
        _orderProcessingTime.Record(durationMs,
            new KeyValuePair<string, object?>("method", method));
    }

    public void UpdateQueueSize(long size)
    {
        Interlocked.Exchange(ref _queueSize, size);
    }
}
```

### 4.1 ASP.NET Core Controller

```csharp
// Controllers/OrdersController.cs

using Microsoft.AspNetCore.Mvc;
using MyApp.Web.Services;
using System.Diagnostics;

namespace MyApp.Web.Controllers;

[ApiController]
[Route("api/[controller]")]
public class OrdersController : ControllerBase
{
    private static readonly ActivitySource ActivitySource = new("MyApp.OrdersController");
    private readonly IOrderService _orderService;
    private readonly ILogger<OrdersController> _logger;

    public OrdersController(IOrderService orderService, ILogger<OrdersController> logger)
    {
        _orderService = orderService;
        _logger = logger;
    }

    [HttpGet("health")]
    public IActionResult HealthCheck()
    {
        return Ok(new { status = "healthy" });
    }

    [HttpPost]
    public async Task<IActionResult> CreateOrder([FromBody] CreateOrderRequest request)
    {
        // ASP.NET Core 已自动创建 Activity
        // 可以获取当前 Activity 添加额外信息
        var currentActivity = Activity.Current;
        currentActivity?.SetTag("http.request.body.size", Request.ContentLength);

        try
        {
            _logger.LogInformation("Creating order for user {UserId}", request.UserId);

            var order = await _orderService.CreateOrderAsync(request.UserId, request.Amount);

            return CreatedAtAction(
                nameof(GetOrder),
                new { orderId = order.OrderId },
                order);
        }
        catch (Exception ex)
        {
            _logger.LogError(ex, "Failed to create order");
            return StatusCode(500, new { error = ex.Message });
        }
    }

    [HttpGet("{orderId}")]
    public async Task<IActionResult> GetOrder(string orderId)
    {
        try
        {
            var order = await _orderService.GetOrderAsync(orderId);

            if (order == null)
            {
                return NotFound(new { error = "Order not found" });
            }

            return Ok(order);
        }
        catch (Exception ex)
        {
            _logger.LogError(ex, "Failed to get order {OrderId}", orderId);
            return StatusCode(500, new { error = ex.Message });
        }
    }

    [HttpPost("bulk")]
    public async Task<IActionResult> ProcessBulkOrders([FromBody] BulkOrderRequest request)
    {
        using var activity = ActivitySource.StartActivity("OrdersController.ProcessBulkOrders");
        activity?.SetTag("order.count", request.OrderIds.Count);

        try
        {
            var results = await _orderService.ProcessBulkOrdersAsync(request.OrderIds);

            return Ok(new { results, count = results.Count });
        }
        catch (Exception ex)
        {
            activity?.RecordException(ex);
            return StatusCode(500, new { error = ex.Message });
        }
    }

    [HttpGet("call-downstream")]
    public async Task<IActionResult> CallDownstream()
    {
        using var httpClient = new HttpClient();

        // HttpClient 的调用会自动传播 Trace 上下文
        var response = await httpClient.GetAsync("http://downstream-service/api/data");
        var content = await response.Content.ReadAsStringAsync();

        return Ok(new { downstreamResponse = content });
    }
}

}
```

### 4.2 自定义中间件

```csharp
// Middleware/RequestTimingMiddleware.cs

using System.Diagnostics;

namespace MyApp.Web.Middleware;

public class RequestTimingMiddleware
{
    private readonly RequestDelegate _next;
    private readonly ILogger<RequestTimingMiddleware> _logger;

    public RequestTimingMiddleware(RequestDelegate next, ILogger<RequestTimingMiddleware> logger)
    {
        _next = next;
        _logger = logger;
    }

    public async Task InvokeAsync(HttpContext context)
    {
        var stopwatch = Stopwatch.StartNew();
        var originalBodyStream = context.Response.Body;

        using var responseBody = new MemoryStream();
        context.Response.Body = responseBody;

        try
        {
            await _next(context);

            stopwatch.Stop();

            // 记录请求时间
            var activity = Activity.Current;
            activity?.SetTag("request.duration_ms", stopwatch.ElapsedMilliseconds);

            // 复制响应到原始流
            responseBody.Position = 0;
            await responseBody.CopyToAsync(originalBodyStream);

            _logger.LogInformation(
                "Request {Method} {Path} completed in {DurationMs}ms - Status {StatusCode}",
                context.Request.Method,
                context.Request.Path,
                stopwatch.ElapsedMilliseconds,
                context.Response.StatusCode);
        }
        catch (Exception ex)
        {
            stopwatch.Stop();

            var activity = Activity.Current;
            activity?.RecordException(ex);
            activity?.SetTag("request.duration_ms", stopwatch.ElapsedMilliseconds);
            activity?.SetStatus(ActivityStatusCode.Error, ex.Message);

            throw;
        }
        finally
        {
            context.Response.Body = originalBodyStream;
        }
    }
}

// 扩展方法
public static class RequestTimingMiddlewareExtensions
{
    public static IApplicationBuilder UseRequestTiming(this IApplicationBuilder builder)
    {
        return builder.UseMiddleware<RequestTimingMiddleware>();
    }
}
```

```csharp
// Telemetry/SamplingConfig.cs

using OpenTelemetry.Trace;
using System.Diagnostics;

namespace MyApp.Web.Telemetry;

public class CustomSampler : Sampler
{
    private readonly TraceIdRatioBasedSampler _baseSampler;

    public CustomSampler(double ratio)
    {
        _baseSampler = new TraceIdRatioBasedSampler(ratio);
    }

    public override SamplingResult ShouldSample(in SamplingParameters samplingParameters)
    {
        var activityName = samplingParameters.Name;

        // 对健康检查不采样
        if (activityName.Contains("health", StringComparison.OrdinalIgnoreCase))
        {
            return new SamplingResult(SamplingDecision.Drop);
        }

        // 对错误全量采样
        if (activityName.Contains("error", StringComparison.OrdinalIgnoreCase) ||
            activityName.Contains("exception", StringComparison.OrdinalIgnoreCase))
        {
            return new SamplingResult(SamplingDecision.RecordAndSample);
        }

        // 对关键路径全量采样
        if (activityName.Contains("payment", StringComparison.OrdinalIgnoreCase) ||
            activityName.Contains("order/create", StringComparison.OrdinalIgnoreCase))
        {
            return new SamplingResult(SamplingDecision.RecordAndSample);
        }

        // 其他按基础采样率
        return _baseSampler.ShouldSample(samplingParameters);
    }
}

// 在 Program.cs 中使用
builder.Services.AddOpenTelemetry()
    .WithTracing(tracing =>
    {
        tracing.SetSampler(new CustomSampler(0.1)); // 10%基础采样率
    });
```

### 5.2 优雅关闭

```csharp
// Telemetry/GracefulShutdownService.cs

public class GracefulShutdownService : IHostedService
{
    private readonly TracerProvider _tracerProvider;
    private readonly MeterProvider _meterProvider;
    private readonly ILogger<GracefulShutdownService> _logger;

    public GracefulShutdownService(
        TracerProvider tracerProvider,
        MeterProvider meterProvider,
        ILogger<GracefulShutdownService> logger)
    {
        _tracerProvider = tracerProvider;
        _meterProvider = meterProvider;
        _logger = logger;
    }

    public Task StartAsync(CancellationToken cancellationToken)
    {
        // 注册关闭事件
        AppDomain.CurrentDomain.ProcessExit += OnProcessExit;
        Console.CancelKeyPress += OnCancelKeyPress;

        return Task.CompletedTask;
    }

    public Task StopAsync(CancellationToken cancellationToken)
    {
        return ShutdownAsync();
    }

    private async void OnProcessExit(object? sender, EventArgs e)
    {
        await ShutdownAsync();
    }

    private async void OnCancelKeyPress(object? sender, ConsoleCancelEventArgs e)
    {
        e.Cancel = true;
        await ShutdownAsync();
    }

    private async Task ShutdownAsync()
    {
        _logger.LogInformation("Shutting down OpenTelemetry...");

        try
        {
            // 强制刷新剩余的Spans
            await _tracerProvider.ForceFlushAsync(CancellationToken.None);
            _tracerProvider.Dispose();

            _logger.LogInformation("TracerProvider shut down successfully");
        }
        catch (Exception ex)
        {
            _logger.LogError(ex, "Error shutting down TracerProvider");
        }

        try
        {
            _meterProvider.Dispose();
            _logger.LogInformation("MeterProvider shut down successfully");
        }
        catch (Exception ex)
        {
            _logger.LogError(ex, "Error shutting down MeterProvider");
        }
    }
}

// 在 Program.cs 中注册
builder.Services.AddSingleton<GracefulShutdownService>();
builder.Services.AddHostedService(provider => provider.GetRequiredService<GracefulShutdownService>());
```
